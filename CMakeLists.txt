cmake_minimum_required(VERSION 2.8.12)
project(darknet_ros)

if(CMAKE_SIZEOF_VOID_P EQUAL 8) 
    set(arch_64 TRUE) 
else() 
    set(arch_64 FALSE) 
endif() 

find_package(CUDA REQUIRED)
find_package(OpenCV REQUIRED)
find_package(catkin REQUIRED COMPONENTS
  cv_bridge
  roscpp
  rospy
  std_msgs
  image_transport
)

catkin_package()

if(arch_64)
   set(
      CUDA_NVCC_FLAGS
      ${CUDA_NVCC_FLAGS};
      -O3
      )
else()
   set(
      CUDA_NVCC_FLAGS
      ${CUDA_NVCC_FLAGS};
      -O3 -gencode arch=compute_53,code=sm_53
      )
endif()

add_definitions(-DGPU -DOPENCV)

include_directories(
  include
  $ENV{CUDA_HOME}/include
  src
  ${catkin_INCLUDE_DIRS}
)

if(arch_64)
  link_directories(
    $ENV{CUDA_HOME}/lib64
  )
else()
  link_directories(
    $ENV{CUDA_HOME}/lib
  )
endif()
